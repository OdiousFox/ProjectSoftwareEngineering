17class SI7006A20:
18 """ class for handling the temperature sensor SI7006-A20
19 +/- 1 deg C error for temperature
20 +/- 5% error for relative humidity
21 datasheet available at https://www.silabs.com/documents/public/data-sheets/Si7006-A20.pdf """
22
23 SI7006A20_I2C_ADDR = const(0x40)
24
25 TEMP_NOHOLDMASTER = const(0xF3)
26 HUMD_NOHOLDMASTER = const(0xF5)

28 def __init__(self, pysense = None, sda = 'P22', scl = 'P21'):
29 if pysense is not None:
30 self.i2c = pysense.i2c
31 else:
32 self.i2c = I2C(0, mode=I2C.MASTER, pins=(sda, scl))
33
34 def _getWord(self, high, low):
35 return ((high & 0xFF) << 8) + (low & 0xFF)
36
37 def temperature(self):
38 """ obtaining the temperature(degrees Celsius) measured by sensor """
39 self.i2c.writeto(SI7006A20_I2C_ADDR, bytearray([0xF3]))
40 time.sleep(0.5)
41 data = self.i2c.readfrom(SI7006A20_I2C_ADDR, 3)
42 #print("CRC Raw temp data: " + hex(data[0]*65536 + data[1]*256 + data[2]))
43 data = self._getWord(data[0], data[1])
44 temp = ((175.72 * data) / 65536.0) - 46.85
45 return temp
46
47 def humidity(self):
48 """ obtaining the relative humidity(%) measured by sensor """
49 self.i2c.writeto(SI7006A20_I2C_ADDR, bytearray([0xF5]))
50 time.sleep(0.5)
51 data = self.i2c.readfrom(SI7006A20_I2C_ADDR, 2)
52 data = self._getWord(data[0], data[1])
53 humidity = ((125.0 * data) / 65536.0) - 6.0
54 return humidity

70 def read_electronic_id(self):
71 """ reading electronic identifier """

100 def dew_point(self):
101 """ computing the dew pointe temperature (deg C) for the current Temperature and Humidity measured pair
102 at dew-point temperature the relative humidity is 100% """
103 temp = self.temperature()
104 humid = self.humidity()
105 h = (math.log(humid, 10) - 2) / 0.4343 + (17.62 * temp) / (243.12 + temp)
106 dew_p = 243.12 * h / (17.62 - h)
107 return dew_p
108
109 def humid_ambient(self, t_ambient, dew_p = None):
110 """ returns the relative humidity compensated for the current Ambient temperature
111 for ex: T-Ambient is 24.4 degC, but sensor indicates Temperature = 31.65 degC and Humidity = 47.3%
112 -> then the actual Relative Humidity is 72.2%
113 this is computed because the dew-point should be the same """
114 if dew_p is None:
115 dew_p = self.dew_point()
116 h = 17.62 * dew_p / (243.12 + dew_p)
117 h_ambient = math.pow(10, (h - (17.62 * t_ambient) / (243.12 + t_ambient)) * 0.4343 + 2)
118 return h_ambient
