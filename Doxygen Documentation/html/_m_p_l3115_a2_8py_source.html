103 def altitude(self):
104 if self.mode == PRESSURE:
105 raise MPL3115A2exception("Incorrect Measurement Mode MPL3115A2")
106
107 OUT_P_MSB = self.i2c.readfrom_mem(MPL3115_I2CADDR, MPL3115_PRESSURE_DATA_MSB,1)
108 OUT_P_CSB = self.i2c.readfrom_mem(MPL3115_I2CADDR, MPL3115_PRESSURE_DATA_CSB,1)
109 OUT_P_LSB = self.i2c.readfrom_mem(MPL3115_I2CADDR, MPL3115_PRESSURE_DATA_LSB,1)
110
111 alt_int = (OUT_P_MSB[0] << 8) + (OUT_P_CSB[0])
112 alt_frac = ((OUT_P_LSB[0] >> 4) & 0x0F)
113
114 if alt_int > 32767:
115 alt_int -= 65536
116
117 return float(alt_int + alt_frac / 16.0)
